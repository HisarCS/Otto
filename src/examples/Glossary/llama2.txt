// ========== SHARED PARAMETERS ==========
param materialThickness 30

// ========== PIECE 2: CIRCLE WITH FOUR RECTANGULAR CUTOUTS ==========
// ---------- Parameters ----------
param circleRadius 100
param rectWidth materialThickness
param rectHeight 60

// ---------- Main Circle ----------
shape circle mainCircle {
  position: [0, 0]
  radius: circleRadius
  fill: false
}

// ---------- Four Rectangular Cutouts (Top, Bottom, Left, Right) ----------
// Top cutout
shape rectangle cutoutTop {
  position: [0, circleRadius - rectHeight/2]
  width: rectWidth
  height: rectHeight
  fill: false
}

// Bottom cutout
shape rectangle cutoutBottom {
  position: [0, -circleRadius + rectHeight/2]
  width: rectWidth
  height: rectHeight
  fill: false
}

// Left cutout
shape rectangle cutoutLeft2 {
  position: [-circleRadius + rectHeight/2, 0]
  rotation: 90
  width: rectWidth
  height: rectHeight
  fill: false
}

// Right cutout
shape rectangle cutoutRight2 {
  position: [circleRadius - rectHeight/2, 0]
  rotation: 90
  width: rectWidth
  height: rectHeight
  fill: false
}

// ---------- Apply All Cutouts Using Difference ----------
difference step1_c {
  add mainCircle
  add cutoutTop
}

difference step2_c {
  add step1_c
  add cutoutBottom
}

difference step3_c {
  add step2_c
  add cutoutLeft2
}

difference llama2 {
  add step3_c
  add cutoutRight2
}
